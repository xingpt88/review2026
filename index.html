    <script>

        // 自动提醒逻辑：如果周日晚上未填写，发送通知

        if ("Notification" in window) {

            Notification.requestPermission();

        function saveReview() {

            const data = {

                date: new Date().toLocaleDateString(),

                video: document.getElementById('videoStats').value,

                followers: document.getElementById('newFollowers').value,

                revenue: document.getElementById('revenue').value,

                health: document.getElementById('healthStats').value,

                reflection: document.getElementById('reflection').value,

                timestamp: Date.now()

            };


            let history = JSON.parse(localStorage.getItem('review2026') || '[]');

            history.unshift(data);

            localStorage.setItem('review2026', JSON.stringify(history));

            

            alert('复盘已保存');

            renderHistory();

            clearForm();

        }


        function renderHistory() {

            const history = JSON.parse(localStorage.getItem('review2026') || '[]');

            const container = document.getElementById('historyList');

            container.innerHTML = history.map(item => `

                <div class="border-b pb-3">

                    <div class="flex justify-between text-xs font-bold text-slate-400 mb-1">

                        <span>${item.date}</span>

                        <span class="text-green-600">$${item.revenue}</span>

                    </div>

                    <div class="text-sm text-slate-700">视频: ${item.video} | 增粉: ${item.followers} | 精力: ${item.health}</div>

                    <div class="text-xs text-slate-500 mt-1 bg-slate-50 p-2 rounded">${item.reflection}</div>

                </div>

            `).join('');

        }


        function clearForm() {

            ['videoStats', 'newFollowers', 'revenue', 'health', 'reflection'].forEach(id => {

                document.getElementById(id).value = '';

            });

        }


        function exportData() {

            const history = localStorage.getItem('review2026');

            const blob = new Blob([history], {type: 'application/json'});

            const url = URL.createObjectURL(blob);

            const a = document.createElement('a');

            a.href = url;

            a.download = '2026-review-data.json';

            a.click();

        }


        // 初始化渲染

        renderHistory();
